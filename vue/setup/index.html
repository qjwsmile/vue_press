提示

在script里直接引用setup，不需要像3.x import vue 导出

<script setup lang="ts">
  import { ref } from "vue";
  const message = ref('hello word');
</script>
<template>
  <div>{{message}}</div>
</template>
<style lang="scss" scoped>
  div{
    text-align: center;
  }
</style>
